def classify_cvd_risk( has_multiple_cvd_events=False, has_acs=False, has_atherosclerosis=False, has_tia_stroke=False, has_zpa_or_aneurysm=False, has_diabetes=False, diabetes_duration_years=0, has_organ_damage=False, has_severe_familial_dyslipidemia=False, ldl_level=0.0, has_systolic_bp_over_180=False, has_tg_over_4_5=False, age=0, score2_percent=0.0 ): # Экстремальный риск if has_multiple_cvd_events and ldl_level > 1.4: return "Экстремальный риск", "<1,0 ммоль/л" # Очень высокий риск if any([ has_acs, has_atherosclerosis, has_tia_stroke, has_zpa_or_aneurysm, (has_diabetes and (has_organ_damage or diabetes_duration_years >= 20)), has_severe_familial_dyslipidemia, (ldl_level >= 5.0) ]): return "Очень высокий риск", "<1,4 ммоль/л" # Высокий риск if any([ has_diabetes and not has_organ_damage and diabetes_duration_years < 20, has_systolic_bp_over_180, has_tg_over_4_5, (ldl_level >= 4.9), (age < 50 and score2_percent > 7.5), (50 <= age < 70 and score2_percent > 10), (age >= 70 and score2_percent > 15) ]): return "Высокий риск", "<1,8 ммоль/л" # Умеренный риск if any([ (age < 50 and 2.5 <= score2_percent <= 7.5), (50 <= age < 70 and 5 <= score2_percent <= 10), (age >= 70 and 7.5 <= score2_percent <= 15) ]): return "Умеренный риск", "<2,6 ммоль/л" # Низкий риск return "Низкий риск", "<3,0 ммоль/л" # Пример вызова risk_category, target_ldl = classify_cvd_risk( has_acs=True, ldl_level=3.2 ) print(f"Категория риска: {risk_category}, Целевой ХС ЛПНП: {target_ldl}")
